### 2.1 潜在 Bug

#### 问题 1：parseInt 对 "0" 值的处理错误

**文件**：settings.html
**行号**：952-968
**问题描述**：

```javascript
// 当前代码
fixedX: parseInt(fixedXInput.value) || DEFAULT_SETTINGS.fixedX;
```

当输入值为 "0" 时，`parseInt("0")` 返回 `0`，`0 || DEFAULT` 会错误地使用默认值。

**修复方式**：

```javascript
// 修复后
fixedX: parseInt(fixedXInput.value) ?? DEFAULT_SETTINGS.fixedX;
```

**影响范围**：

- settings.html 第 952-968 行（getSettings 函数）
- settings.html 第 1004-1020 行（saveSettings 函数）
- index.html 第 816 行

#### 问题 10：根据引脚朝向确定导线方向逻辑重复

**问题描述**：

**修复方式**：

```javascript
// 提取为独立函数
function getWireLineFromPin(pin, wireLength = 50) {
	const rotation = pin.rotation || 0;

	if (rotation >= 315 || rotation < 45) {
		// 向右 (0°)
		return [pin.x, pin.y, pin.x + wireLength, pin.y];
	} else if (rotation >= 45 && rotation < 135) {
		// 向下 (90°)
		return [pin.x, pin.y, pin.x, pin.y + wireLength];
	} else if (rotation >= 135 && rotation < 225) {
		// 向左 (180°)
		return [pin.x, pin.y, pin.x - wireLength, pin.y];
	} else {
		// 向上 (270°)
		return [pin.x, pin.y, pin.x, pin.y - wireLength];
	}
}

// 使用
const line = getWireLineFromPin(pin, settings.wireLength || 50);
```

### 3.2 性能问题

#### 问题 12：updatePowerPreview 触发过于频繁

**文件**：settings.html
**行号**：680-685
**问题描述**：
每次输入都触发预览更新，可能造成性能问题。

**修复方式**：

```javascript
// 添加防抖处理
let powerPreviewTimeout = null;

function updatePowerPreview() {
	clearTimeout(powerPreviewTimeout);
	powerPreviewTimeout = setTimeout(() => {
		const name = $('newPowerName')
			.value.toUpperCase()
			.replace(/[^A-Z0-9]/g, '');
		const voltage = $('newPowerVoltage')
			.value.toUpperCase()
			.replace(/[^0-9V]/g, '');

		if (name && voltage) {
			$('newPowerPreview').innerHTML = `<span class="text-danger">PW_${name}_${voltage}</span>`;
		} else {
			$('newPowerPreview').innerHTML = '<span class="text-muted">PW_名称_电压</span>';
		}
	}, 150); // 150ms 防抖延迟
}
```
